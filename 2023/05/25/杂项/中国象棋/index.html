中国象棋

花了三个晚上写的小游戏。勉强能玩。

我编写的程序里，AI 只能看自己的一步棋，所以很弱。但是看到他被我捉子会躲，被将军会垫子，还是产生了巨大的成就感。

这么着，就决定发到博客里了。

后来又花了一个晚上重写了部分代码，各方面都有进步。

一些说明

  • 使用鼠标点击棋子进行操作

    在任何情况下，点击我方棋子即选中该棋子。

    被选中的棋子被品红色圆圈指出，其移动范围被白点标记，攻击范围被青色点标记。

    选中棋子的情况下，点击要移动的格子即完成移动，按右键或点击空白处则取消选中。

  • 键盘 E 键：红黑交换（不可用）

    键盘 L 键：投降

    键盘 O 键：直接获胜

// 开始游戏的方法：执行 new ChessFrame();


class ChessFrame extends JFrame {

    private int windowWidth = 900; //600;
    private int windowHeight = 1020; //680;
    private int cellEdge = 90; //60;
    private GamePanel gamePanel = new GamePanel(this);

    ChessFrame() {
        setSize(windowWidth, windowHeight);
        setVisible(true);
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        add(gamePanel);
        addKeyListener(gamePanel);
        addMouseListener(gamePanel);
        new Thread(gamePanel).start();
    }

    public int getCellEdge() {
        return cellEdge;
    }

    public int getWindowWidth() {
        return windowWidth;
    }

    public int getWindowHeight() {
        return windowHeight;
    }
}

class GamePanel extends JPanel implements Runnable, KeyListener, MouseListener {
    public static final char NONE_FLAG = '\u0000';
    public static final Color CHESS_FILL = new Color(240, 170, 70);
    private static String init = "cnxsksxnc          r     r p p p p p                  P P P P P R     R          CNXSKSXNC";


    private int cellEdge;
    private int ori;
    private final ChessFrame root;
    private char[] chessboard = new char[90];
    public Player p1 = new Player(1, this);     // 电脑
    public Player p2 = new Player(0, this);     // 玩家
    private int selected = -1;
    private ArrayList<int[]> selectedRange = null;
    private boolean playerCheesing = true;
    private int end = 0;


    GamePanel(ChessFrame cf) {
        this.root = cf;
        cellEdge = cf.getCellEdge();
        ori = cellEdge >> 1;
    }

    @Override
    public void run() {
        initial();
        while (true) {
            repaint();
            if (end == 0) {
                if (p1.lose) end |= 1;
                if (p2.lose) end |= 2;
            }
            if (end == 0 && !playerCheesing) {
                int[] analyse = AnalyseHelper.analyse(boardToString(), p1.getKing(), p2.getKing(), 4);
                move(analyse[0], analyse[1]);
                playerCheesing = true;
            }
        }
    }

    private void initial() {
        chessboard = init.replace(' ', '\u0000').toCharArray();
    }

    @Override
    public void paint(Graphics g) {
        drawBoard(g);
        drawPieces(g);
        drawSelectedPiece(g);
        if (end > 0) drawEndGame(g, end == 1);
    }

    private void drawEndGame(Graphics g, boolean win) {
        g.setColor(win ? Color.WHITE: Color.LIGHT_GRAY);
        g.fillRect(0, (root.getWindowHeight()>>1)-((cellEdge+3*ori)>>1), root.getWindowWidth(), ori);
        g.setFont(new Font("华文行楷", Font.BOLD, (3*ori>>1)));
        String out = win ? "技高一筹" : "棋差一着";
        Color c = win ? new Color(200, 50, 50) : new Color(30, 30, 50);
        g.setColor(c);
        g.drawString(out, root.getWindowWidth()>>1, (root.getWindowHeight()>>1)-cellEdge);
    }

    private void drawBoard(Graphics g) {
        g.setColor(Color.ORANGE);
        g.fillRect(0, 0, cellEdge * 9, cellEdge * 10);
        g.setColor(Color.BLACK);
        g.drawRect(ori, ori, cellEdge * 8, cellEdge * 9);
        g.drawRect(ori - (ori >> 3), ori - (ori >> 3), cellEdge * 8 + (ori >> 3) * 2, cellEdge * 9 + (ori >> 3) * 2);
        for (int i = 1; i <= 8; i++) {
            g.drawLine(ori, ori + cellEdge * i, ori + 8 * cellEdge, ori + cellEdge * i);
        }
        for (int i = 1; i <= 7; i++) {
            g.drawLine(ori + cellEdge * i, ori, ori + cellEdge * i, ori + cellEdge * 4);
            g.drawLine(ori + cellEdge * i, ori + cellEdge * 5, ori + cellEdge * i, ori + cellEdge * 9);
        }
        g.drawLine(ori + 3 * cellEdge, ori, ori + 5 * cellEdge, ori + 2 * cellEdge);
        g.drawLine(ori + 3 * cellEdge, ori + 2 * cellEdge, ori + 5 * cellEdge, ori);
        g.drawLine(ori + 3 * cellEdge, ori + 7 * cellEdge, ori + 5 * cellEdge, ori + 9 * cellEdge);
        g.drawLine(ori + 3 * cellEdge, ori + 9 * cellEdge, ori + 5 * cellEdge, ori + 7 * cellEdge);
        for (int i = 0; i < 5; i++) {
            int dir = (i == 0 ? 0 : 1) | (i == 4 ? 0 : 2);
            drawBroadALine(i * 2, 3, dir, g);
            drawBroadALine(i * 2, 6, dir, g);
        }
        drawBroadALine(1, 2, 3, g);
        drawBroadALine(7, 2, 3, g);
        drawBroadALine(1, 7, 3, g);
        drawBroadALine(7, 7, 3, g);
    }

    private void drawBroadALine(int x, int y, int dir, Graphics g) {
        int small = cellEdge >> 3;
        int small2 = small * 2;
        if ((dir & 1) > 0) {
            g.drawLine(ori + x * cellEdge - small, ori + y * cellEdge - small2, ori + x * cellEdge - small, ori + y * cellEdge - small);
            g.drawLine(ori + x * cellEdge - small2, ori + y * cellEdge - small, ori + x * cellEdge - small, ori + y * cellEdge - small);
            g.drawLine(ori + x * cellEdge - small, ori + y * cellEdge + small2, ori + x * cellEdge - small, ori + y * cellEdge + small);
            g.drawLine(ori + x * cellEdge - small2, ori + y * cellEdge + small, ori + x * cellEdge - small, ori + y * cellEdge + small);
        }
        if ((dir & 2) > 0) {
            g.drawLine(ori + x * cellEdge + small, ori + y * cellEdge - small2, ori + x * cellEdge + small, ori + y * cellEdge - small);
            g.drawLine(ori + x * cellEdge + small2, ori + y * cellEdge - small, ori + x * cellEdge + small, ori + y * cellEdge - small);
            g.drawLine(ori + x * cellEdge + small, ori + y * cellEdge + small2, ori + x * cellEdge + small, ori + y * cellEdge + small);
            g.drawLine(ori + x * cellEdge + small2, ori + y * cellEdge + small, ori + x * cellEdge + small, ori + y * cellEdge + small);
        }
    }

    private void drawPieces(Graphics g) {
        for (int t = 0; t < 90; t++) {
            char piece = chessboard[t];
            if (piece != '\u0000') drawSingleChess(g, t%9, t/9, piece, Character.isUpperCase(piece) ? p2.color : p1.color);
        }
    }

    private void drawSingleChess(Graphics g, int x, int y, char species, Color cc) {
        int chessLength = (cellEdge >> 2) + (cellEdge >> 3) + (cellEdge >> 4);
        int outRing = (int) (chessLength * 0.83);
        int half_fontsize = (int) (chessLength * 0.707 * 0.9);
        g.setColor(CHESS_FILL);
        g.fillOval(ori + x * cellEdge - chessLength, ori + y * cellEdge - chessLength, chessLength * 2, chessLength * 2);
        g.setColor(Color.BLACK);
        g.drawOval(ori + x * cellEdge - chessLength, ori + y * cellEdge - chessLength, chessLength * 2, chessLength * 2);
        g.drawOval(ori + x * cellEdge - outRing, ori + y * cellEdge - outRing, outRing * 2, outRing * 2);
        g.setColor(cc);
        g.setFont(new Font("楷体", Font.BOLD, half_fontsize * 2));
        g.drawString(Piece.chineseCharacter.getOrDefault(species, ' ') + "", ori + x * cellEdge - half_fontsize, ori + y * cellEdge + half_fontsize);
    }

    private void drawSelectedPiece(Graphics g) {
        if (selectedRange == null || selected == -1) return;
        int bullSize = cellEdge >> 3;
        char selectedPiece = chessboard[selected];
        if (selectedPiece == '\u0000') return;
        for (int[] ints : selectedRange) {
            int x = ints[0], y = ints[1];
            if (ints[2] == 2) {
                g.setColor(Color.MAGENTA);
                g.fillOval(ori + x * cellEdge - (cellEdge >> 1), ori + y * cellEdge - (cellEdge >> 1), cellEdge, cellEdge);
            } else {
                if (ints[2] == 1) g.setColor(Color.CYAN);
                else g.setColor(Color.WHITE);
                g.fillOval(ori + x * cellEdge - bullSize, ori + y * cellEdge - bullSize, bullSize * 2, bullSize * 2);
            }
        }
        drawSingleChess(g, selected%9, selected/9, selectedPiece, Character.isUpperCase(selectedPiece) ? p2.color : p1.color);
    }


    public char checkPos(int p) {
        if (p < 0 || p > 89) return '\u0000';
        return chessboard[p];
    }

    public void move(int ori, int np) {
        if (chessboard[np] == 'k') p1.lose = true;
        else if (chessboard[np] == 'K') p2.lose = true;
        chessboard[np] = chessboard[ori];
        chessboard[ori] = '\u0000';
    }

    public String boardToString() {
        return new String(chessboard);
    }

    private void setSelected(int p) {
        selected = p;
        if (selected < 0 || selected > 89 || chessboard[selected] == '\u0000') {
            selectedRange = null;
        } else if (Character.isUpperCase(chessboard[selected])) {
            selectedRange = Piece.getRange(boardToString(), p);
        } else {
            selectedRange = new ArrayList<>();
            selectedRange.add(new int[]{p%9, p/9, 2});
        }
    }

    private void reverse() {
        Player temp = p1;
        p1 = p2;
        p2 = temp;
        for (int t = 0; t < 45; t++) {
            for (int x = 0; x < 9; x++) {
                char c = chessboard[t];
                chessboard[t] = chessboard[89-t];
                chessboard[89-t] = chessboard[t];
            }
        }
        playerCheesing = !playerCheesing;
    }

    @Override
    public void keyTyped(KeyEvent e) {
    }

    @Override
    public void keyPressed(KeyEvent e) {

    }

    @Override
    public void keyReleased(KeyEvent e) {
        switch (e.getKeyCode()) {
            case KeyEvent.VK_E:
                reverse();
                break;
            case KeyEvent.VK_L:
                p2.lose = true;
                break;
            case KeyEvent.VK_O:
                p1.lose = true;
                break;
        }
    }

    @Override
    public void mouseClicked(MouseEvent e) {
        if (end > 0 || e.getButton() != MouseEvent.BUTTON1) {
            setSelected(-1);
            return;
        }
        int y = (e.getY()-ori)/cellEdge, x = (e.getX()-ori)/cellEdge;
        if (selected >= 0 && selectedRange != null) {
            for (int[] points: selectedRange) {
                if (points[2] == 2) continue;
                int px = points[0], py = points[1];
                if (px == x && py == y) {
                    move(selected, py*9+px);
                    setSelected(-1);
                    playerCheesing = false;
                    return;
                }
            }
        }
        if (y >= 0 && x >= 0 && y < 10 && x < 9) setSelected(y*9+x);
    }

    @Override
    public void mousePressed(MouseEvent e) {
    }

    @Override
    public void mouseReleased(MouseEvent e) {
    }

    @Override
    public void mouseEntered(MouseEvent e) {

    }

    @Override
    public void mouseExited(MouseEvent e) {

    }
}

class Player {
    private static final int CHANGE = 32;
    private char king;
    public boolean lose = false;
    private final GamePanel broad;
    public final Color color;

    public Player(int p, GamePanel broad) {
        this.broad = broad;
        if ((p & 1) == 0) {
            color = new Color(200, 0, 0);
            king = 'K';
        } else {
            color = new Color(0, 100, 0);
            king = 'k';
        }
    }

    public GamePanel getBroad() {
        return broad;
    }

    public char getKing() {
        return king;
    }

}

class Piece {
    // 以下各标记，大写为红方，小写为黑方
    public static final char K = 'K';      // 帥将
    public static final char C = 'C';      // 俥車
    public static final char N = 'N';      // 傌馬
    public static final char P = 'P';      // 兵卒
    public static final char X = 'X';      // 相象
    public static final char R = 'R';      // 炮砲
    public static final char S = 'S';      // 仕士
    public static final Map<Character, Character> chineseCharacter = new HashMap<>();
    private static final char NONE = '\u0000';

    static {
        chineseCharacter.put('K', '帥');
        chineseCharacter.put('k', '将');
        chineseCharacter.put('C', '俥');
        chineseCharacter.put('c', '車');
        chineseCharacter.put('N', '傌');
        chineseCharacter.put('n', '馬');
        chineseCharacter.put('P', '兵');
        chineseCharacter.put('p', '卒');
        chineseCharacter.put('X', '相');
        chineseCharacter.put('x', '象');
        chineseCharacter.put('R', '炮');
        chineseCharacter.put('r', '砲');
        chineseCharacter.put('S', '仕');
        chineseCharacter.put('s', '士');
    }

    public static ArrayList<int[]> getRange(String state, int p) {
        switch (state.charAt(p)) {
            case 'K':
            case 'k':
                return getKRange(state, p);
            case 'C':
            case 'c':
                return getCRange(state, p);
            case 'N':
            case 'n':
                return getNRange(state, p);
            case 'P':
            case 'p':
                return getPRange(state, p);
            case 'X':
            case 'x':
                return getXRange(state, p);
            case 'R':
            case 'r':
                return getRRange(state, p);
            case 'S':
            case 's':
                return getSRange(state, p);
            default:
                return null;
        }
    }

    private static ArrayList<int[]> getKRange(String state, int p) {
        int y = p / 9, x = p % 9;
        ArrayList<int[]> res = new ArrayList<>();
        if (x > 3) checkToAdd(state, p, x - 1, y, res);
        if (x < 5) checkToAdd(state, p, x + 1, y, res);
        if ((y > 0 && y < 3) || (y > 7)) checkToAdd(state, p, x, y - 1, res);
        if ((y < 2) || (y > 6 && y < 9)) checkToAdd(state, p, x, y + 1, res);
        res.add(new int[]{x, y, 2});
        return res;
    }

    private static ArrayList<int[]> getCRange(String state, int p) {
        int y = p / 9, x = p % 9;
        ArrayList<int[]> res = new ArrayList<>();
        for (int i = 1; ; i++) if (checkToAdd(state, p, x + i, y, res)) break;
        for (int i = 1; ; i++) if (checkToAdd(state, p, x - i, y, res)) break;
        for (int i = 1; ; i++) if (checkToAdd(state, p, x, y + i, res)) break;
        for (int i = 1; ; i++) if (checkToAdd(state, p, x, y - i, res)) break;
        res.add(new int[]{x, y, 2});
        return res;
    }

    private static ArrayList<int[]> getNRange(String state, int p) {
        int y = p / 9, x = p % 9;
        ArrayList<int[]> res = new ArrayList<>();
        if (checkPos(x - 1, y) && state.charAt(p - 1) == NONE) {
            checkToAdd(state, p, x - 2, y + 1, res);
            checkToAdd(state, p, x - 2, y - 1, res);
        }
        if (checkPos(x + 1, y) && state.charAt(p + 1) == NONE) {
            checkToAdd(state, p, x + 2, y + 1, res);
            checkToAdd(state, p, x + 2, y - 1, res);
        }
        if (checkPos(x, y - 1) && state.charAt(p - 9) == NONE) {
            checkToAdd(state, p, x + 1, y - 2, res);
            checkToAdd(state, p, x - 1, y - 2, res);
        }
        if (checkPos(x, y + 1) && state.charAt(p + 9) == NONE) {
            checkToAdd(state, p, x - 1, y + 2, res);
            checkToAdd(state, p, x + 1, y + 2, res);
        }
        res.add(new int[]{x, y, 2});
        return res;
    }

    private static ArrayList<int[]> getPRange(String state, int p) {
        int y = p / 9, x = p % 9;
        char c = state.charAt(p);
        ArrayList<int[]> res = new ArrayList<>();
        if (Character.isLowerCase(c)) {
            if (y < 5) checkToAdd(state, p, x, y + 1, res);
            else {
                checkToAdd(state, p, x - 1, y, res);
                checkToAdd(state, p, x + 1, y, res);
                checkToAdd(state, p, x, y + 1, res);
            }
        } else {
            if (y > 4) checkToAdd(state, p, x, y - 1, res);
            else {
                checkToAdd(state, p, x - 1, y, res);
                checkToAdd(state, p, x + 1, y, res);
                checkToAdd(state, p, x, y - 1, res);
            }
        }
        res.add(new int[]{x, y, 2});
        return res;
    }

    private static ArrayList<int[]> getXRange(String state, int p) {
        int y = p / 9, x = p % 9;
        ArrayList<int[]> res = new ArrayList<>();
        char c = state.charAt(p);
        if (p < 45) {
            if (checkPos(x - 2, y - 2) && state.charAt(p - 10) == NONE) checkToAdd(state, p, x - 2, y - 2, res);
            if (y < 3 && checkPos(x + 2, y + 2) && state.charAt(p + 10) == NONE) checkToAdd(state, p, x + 2, y + 2, res);
            if (checkPos(x + 2, y - 2) && state.charAt(p - 8) == NONE) checkToAdd(state, p, x + 2, y - 2, res);
            if (y < 3 && checkPos(x - 2, y + 2) && state.charAt(p + 8) == NONE) checkToAdd(state, p, x - 2, y + 2, res);
        } else {
            if (checkPos(x - 2, y + 2) && state.charAt(p + 8) == NONE) checkToAdd(state, p, x - 2, y + 2, res);
            if (y > 6 && checkPos(x + 2, y - 2) && state.charAt(p - 8) == NONE) checkToAdd(state, p, x + 2, y - 2, res);
            if (checkPos(x + 2, y + 2) && state.charAt(p + 10) == NONE) checkToAdd(state, p, x + 2, y + 2, res);
            if (y > 6 && checkPos(x - 2, y - 2) && state.charAt(p - 10) == NONE) checkToAdd(state, p, x - 2, y - 2, res);
        }
        res.add(new int[]{x, y, 2});
        return res;
    }

    private static ArrayList<int[]> getRRange(String state, int p) {
        int y = p / 9, x = p % 9;
        ArrayList<int[]> res = new ArrayList<>();
        for (int i = 1, f = 0; x + i < 9; i++) {
            char temp = state.charAt(p + i);
            if (temp != NONE) {
                if (f == 1) {
                    checkToAdd(state, p, x + i, y, res);
                    break;
                }
                f = 1;
            } else if (f == 0) {
                checkToAdd(state, p, x + i, y, res);
            }
        }
        for (int i = 1, f = 0; x - i >= 0; i++) {
            char temp = state.charAt(p - i);
            if (temp != NONE) {
                if (f == 1) {
                    checkToAdd(state, p, x - i, y, res);
                    break;
                }
                f = 1;
            } else if (f == 0) {
                checkToAdd(state, p, x - i, y, res);
            }
        }
        for (int i = 1, f = 0; y + i < 10; i++) {
            char temp = state.charAt(p + 9 * i);
            if (temp != NONE) {
                if (f == 1) {
                    checkToAdd(state, p, x, y + i, res);
                    break;
                }
                f = 1;
            } else if (f == 0) {
                checkToAdd(state, p, x, y + i, res);
            }
        }
        for (int i = 1, f = 0; y - i >= 0; i++) {
            char temp = state.charAt(p - i * 9);
            if (temp != NONE) {
                if (f == 1) {
                    checkToAdd(state, p, x, y - i, res);
                    break;
                }
                f = 1;
            } else if (f == 0) {
                checkToAdd(state, p, x, y - i, res);
            }
        }
        res.add(new int[]{x, y, 2});
        return res;
    }

    private static ArrayList<int[]> getSRange(String state, int p) {
        int y = p / 9, x = p % 9;
        ArrayList<int[]> res = new ArrayList<>();
        boolean up = Character.isUpperCase(state.charAt(p));
        if (x == 4) {
            checkToAdd(state, p, x + 1, y + 1, res);
            checkToAdd(state, p, x - 1, y + 1, res);
            checkToAdd(state, p, x + 1, y - 1, res);
            checkToAdd(state, p, x - 1, y - 1, res);
        } else checkToAdd(state, p, 4, up ? 8 : 1, res);
        res.add(new int[]{x, y, 2});
        return res;
    }

    private static boolean checkToAdd(String state, int p, int x, int y, ArrayList<int[]> res) {
        char s = state.charAt(p);
        if (!checkPos(x, y)) return true;
        char tar = state.charAt(y * 9 + x);
        if (tar == '\u0000') res.add(new int[]{x, y, 0});
        else {
            if (Character.isUpperCase(tar) ^ Character.isUpperCase(s)) res.add(new int[]{x, y, 1});
            return true;
        }
        return false;
    }

    private static boolean checkPos(int x, int y) {
        return x < 9 && x >= 0 && y < 10 && y >= 0;
    }

    private static boolean checkPos(int p) {
        return p >= 0 && p < 90;
    }
}

class AnalyseHelper {
    public static Map<Character, int[][]> pos_value = new HashMap<>();
    public static Map<Character, Integer> value = new HashMap<>();
    private static char NONE = '\u0000';

    static {
        int[][] TEMP = new int[10][5];
        int[][] KV = {{0,0,0,30,40},{0,0,0,20,30},{0,0,0,10,20}};

        value.put('K', 10000);
        int[][] CV = {{-10,0,0,0,-50},{0,0,-10,-20,-40},{-10,0,-20,0,-50}
                ,{0,5,0,10,0},{0,0,0,0,0},{0,20,-5,20,0},{0,20,20,20,20}
                ,{0,30,30,10,5},{0,35,35,50,5},{20,20,20,20,20}};
        pos_value.put('C', TEMP);
        value.put('C', 90);
        int[][] NV = {{-40,-10,0,-30,-50},{-20,-10,0,-35,-50},{-10,-10,0,10,-40}
                ,{-5,5,0,5,0},{0,10,15,10,5},{0,5,10,10,15},{0,25,5,25,10}
                ,{5,5,10,20,0},{0,0,20,-10,-10},{0,0,0,0,-15}};
        pos_value.put('N', TEMP);

        int[][] RV = {{-10,-10,0,0,-20},{-20,-20,-10,-35,-30},{0,0,5,5,30}

    public static int[] analyse(String states, char selfKing, char oppKing, int level) {
        boolean sf = Character.isUpperCase(selfKing);
        int score = Integer.MIN_VALUE >> 1;
        int[] res = new int[3];
        for (int y = 0; y < 9; y++) {
            for (int x = 0; x < 8; x++) {
                int cp = y*9+x;
                char curr = states.charAt(cp);
                if (curr != NONE && Character.isUpperCase(curr) == sf) {
                    ArrayList<int[]> range = Piece.getRange(states, cp);
                    if (range == null || range.size() == 0) continue;
                    for (int[] step: range) {
                        if (step[2] == 2) continue;
                        char[] newState = states.toCharArray();
                        int np = step[1] * 9+step[0];
                        newState[np] = curr;
                        newState[cp] = NONE;
                        int newScore = 0;
                        String ns = new String(newState);
                        if (level > 0) {
                            newScore = -analyse(ns, oppKing, selfKing, level - 1)[2];
                        }
                        else newScore = countScore(ns, selfKing);
                        if (newScore > score) {
                            res[0] = cp;
                            res[1] = np;
                            res[2] = newScore;
                            score = newScore;
                        }
                    }
                }
            }
        }
        return res;
    }

    private static int countScore(String states, char selfKing) {
        boolean rk = Character.isUpperCase(selfKing);
        int currentScore = 0;
        for (int p = 0; p < 90; p++) {
            char c = states.charAt(p);
            if (c == '\u0000') continue;
            int v = getVal(c, p);
            if (Character.isUpperCase(c) ^ rk) currentScore -= v;
            else currentScore += v;
        }
        return currentScore;
    }

    private static int getVal(char piece, int p) {
        if (Character.isUpperCase(piece)) p = 89-p;
        int y = p/9, x = p%9;
        if (x > 4) x = 8-x;
        piece = Character.toUpperCase(piece);
        try {
            return value.get(piece) + pos_value.get(piece)[y][x];
        } catch (Exception e) {
            System.out.println(piece + "\t" + p);
            e.printStackTrace();
            throw e;
        }
    }
